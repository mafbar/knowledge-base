## Parsing Links from a Telegram Chat

I have this Telegram group chat where there are various links constantly being posted, and they have accumulated to over three thousand links over the course of a few years. I wanted a simple way to aggregate the links in a more useful way, like pasting them into a spreadsheet for easier management. Here's what I did:

1. I installed the Telegram Desktop application, and signed in my account there.
2. I went to the target group chat, clicked the hamburger menu on the top right, and clicked "Export chat history".
3. It asked me for the export settings, like including photos, videos, voice messages, files, etc., the date range, and the download path for the chat export. I chose according to what I needed, which was only text.
4. Click the exporting format. I chose to export both, which are Human-readable HTML and Machine-readable JSON. I only used the JSON. You can probably use either, but I think extracting data from JSON files seem to be the more intuitive choice.
5. I put them in a folder. This will be my working directory.
6. I created a Python file named "extracted_links_with_context.tsv". The script will parse the links and their respective contexts (like a descriptor, or article title, or product listing name, etc.), and create a tab-separated value file with two columns, the first containing the context (if it exists), and the second containing the link. The reason it is tab-separated value instead of comma-separated value is because often times, the context, especially article titles, will contain commas, so it will not separate the texts properly. Here is the script:

```python
import json
import csv

# Load the exported Telegram JSON file
with open('result.json', 'r', encoding='utf-8') as f:
    data = json.load(f)

messages = data.get('messages', [])
rows = []

for msg in messages:
    text_content = msg.get('text', [])
    
    if isinstance(text_content, list):
        current_text_context = ""
        
        # Gather plain text context
        for item in text_content:
            if isinstance(item, str):
                current_text_context += item.strip() + " "
        
        # Find the links and pair them
        for item in text_content:
            if isinstance(item, dict) and item.get('type') == 'link':
                url = item.get('text')
                if url:
                    clean_context = current_text_context.strip()
                    if not clean_context:
                        clean_context = "No text description provided"
                        
                    rows.append([clean_context, url])

# Save as a .tsv file using tabs (\t) as the delimiter
with open('extracted_links_with_context.tsv', 'w', encoding='utf-8-sig', newline='') as out_f:
    # QUOTE_MINIMAL automatically puts quotes around any text fields that contain formatting surprises
    writer = csv.writer(out_f, delimiter='\t', quoting=csv.QUOTE_MINIMAL)
    writer.writerow(['Context / Title', 'URL']) # Header rows
    writer.writerows(rows)

print(f"Done! Saved {len(rows)} links to 'extracted_links_with_context.tsv'.")
```
7. It will create a file named extracted_links_with_context.tsv (or you can name it whatever you wish).
8. I then pasted the content of this file to any spreadsheet program, like Google Sheets or Microsoft Excel or LibreOffice Calc.

This script is generated by Gemini Flash 3.5.

## Delete files with a particular string in a directory (Python)

This script will delete all files in a directory containing a particular string.

```python
import os
import argparse

def delete_files_with_string(directory, search_string):
    deleted_count = 0
    # Ensure the directory exists before scanning
    if not os.path.exists(directory):
        print(f"Error: The directory '{directory}' does not exist.")
        return

    for filename in os.listdir(directory):
        if search_string in filename:
            file_path = os.path.join(directory, filename)
            try:
                os.remove(file_path)
                print(f"Deleted: {file_path}")
                deleted_count += 1
            except Exception as e:
                print(f"Failed to delete {file_path}: {e}")
                
    if deleted_count == 0:
        print(f"No files containing '{search_string}' were found to delete.")
    else:
        print(f"Successfully deleted {deleted_count} file(s).")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Delete files containing a specific string.")

    # Positional argument (required)
    parser.add_argument("search_string", type=str, help="The string to look for in filenames")
    
    # Optional argument for directory (defaults to current directory if omitted)
    parser.add_argument("-d", "--directory", type=str, default=".", help="The directory to execute the deletion in (default: current directory)")

    args = parser.parse_args()

    # Execute the deletion using the parsed arguments
    delete_files_with_string(args.directory, args.search_string)
```


## To search a particular string in every filename of a directory (Python)

This script will search a particular string in the filename of every file in a particular directory.

```python
import os
import argparse

def find_files_with_string(directory, search_string):
    files_with_string = []
    for filename in os.listdir(directory):
        if search_string in filename:
            files_with_string.append(filename)
    return files_with_string

if __name__ == "__main__":
    # Set up argument parsing
    parser = argparse.ArgumentParser(description="Search for files in a directory containing a specific string.")
    
    # Add arguments
    parser.add_argument("search_string", type=str, help="The string to look for in filenames.")
    parser.add_argument("-d", "--directory", type=str, default=".", help="The directory to search in (default: current directory).")
    
    # Parse the arguments from the terminal
    args = parser.parse_args()

    # Execute the search
    files_with_string = find_files_with_string(args.directory, args.search_string)

    # Print results
    if files_with_string:
        print(f"Files containing the string '{args.search_string}':")
        for file in files_with_string:
            print(file)
    else:
        print(f"No files found with the string '{args.search_string}'.")
```

To use it, you can launch the terminal and use it like so:

```bash
script.py stringthatyouwant
```

Additionally, you can add the directory in which you want to run the script.

```python

```

## Rename all files in a directory with a number, alongside its original name. (Windows cmd)

This script will rename all the files in a particular directory by prepending all filenames with a particular word, an index number, and its original filename. This is written in Windows batch script.

```bat
@echo off
setlocal enabledelayedexpansion
set /a count=1

for %%a in (*.jpg) do (
    ren "%%a" "vacation_0!count!_%%a"
    set /a count+=1
)
```

For example, if I have:

- asd.jpg
- f22.jpg
- xki.jpg

It will become:

- vacation_01_asd.jpg
- vacation_02_f22.jpg
- vacation_03_xki.jpg

A version of this script that completely rewrites the original filenames with a new naming scheme is as follows:

```bat
@echo off
setlocal enabledelayedexpansion
set /a count=1

for %%a in (*.png) do (
    ren "%%a" "test0!count!.png"
    set /a count+=1
)
```

This script will rename stuff from:

- asldk.png
- sadj2.png
- c90xj.png

to:

- test01.png
- test02.png
- test03.png

## Move all files in the current directory, with the list taken from list.txt, and move them into the test directory. (Windows cmd)

This script will move all wanted files in the working directory to a target directory, and the wanted files are listed in a text file (in this example, it's list.txt). This is written in Windows batch script.

```bat
@echo off
setlocal enabledelayedexpansion

set "dest_dir=test"

for /F "tokens=*" %%f in (list.txt) do (
    copy "%%f" "%dest_dir%\"
)
```

Note: If I don't have the "tokens=\*", then it will simply separate all files by whitespaces. If I do have "tokens=\*", it will separate all files by newlines. So if your list.txt is separated by newlines, use the script above.

## Scans a directory for files, extract the names at the start of the filenames, and organise all files according to their respective names

This script scans the source directory for files. For each filename, it extracts a “group key” by taking the beginning of the name (up to the hyphen separator). It then creates a folder named after that key inside out_base (by default grouped) and copies (or moves) the file into that folder. This way, all pictures that share the same leading person name get collected together automatically.

```python
from pathlib import Path
import shutil
import re

# Delimiter exclusively: hyphen-like characters only
HYPHEN_RE = r"[–—-]"  # en dash, em dash, normal hyphen

def extract_group_key(filename: str) -> str:
    stem = Path(filename).stem
    # Capture from start until first hyphen-like char (or end)
    m = re.match(rf"^\s*(.+?)(?:{HYPHEN_RE}+|$)", stem)
    return m.group(1).strip() if m else stem.strip()

def group_by_leading_key(src_dir: str, out_base: str | None = None, mode: str = "copy"):
    src = Path(src_dir)
    if out_base is None:
        out_base = src_dir

    out_base = Path(out_base)
    out_base.mkdir(parents=True, exist_ok=True)

    for p in src.iterdir():
        if not p.is_file():
            continue

        key = extract_group_key(p.name)
        if not key:
            continue

        target_dir = out_base / key
        target_dir.mkdir(parents=True, exist_ok=True)

        target = target_dir / p.name
        if mode == "move":
            shutil.move(str(p), str(target))
        else:
            shutil.copy2(str(p), str(target))

if __name__ == "__main__":
    group_by_leading_key(
        src_dir=".",
        out_base="grouped",
        mode="copy",  # or "move"
    )
```
So if I have lots of files starting with john, hoho, and pox, it will go like so:

- john - nature 1... → captures john
- hoho - trip 3 → captures hoho
- pox - ... → captures pox

And then, all files starting with john will be put in a folder. Same goes with hoho and pox. This script could probably be modified by an LLM to include (or exclude) all sorts of delimiters.

## Move a content in a folder to the directory directly above it, and delete the folder

The script takes folder names from the command line (you list them as arguments). For each folder, it checks that it actually exists and is a directory. Then it moves every item inside that folder into your current working directory (the directory where you run the script), leaving the original folder name behind. Finally, it tries to delete the now-empty folder; if the folder isn’t empty or can’t be removed, it quietly leaves it.

```python
from pathlib import Path
import shutil
import sys

def move_folder_contents(folder: Path, dst_dir: Path):
    if not folder.exists() or not folder.is_dir():
        print(f"Skipping (not a folder): {folder}")
        return

    # Move everything inside the folder (not the folder itself)
    for item in folder.iterdir():
        target = dst_dir / item.name
        if target.exists():
            print(f"Skipping (already exists): {target}")
            continue
        shutil.move(str(item), str(target))
    # Optional: remove the now-empty folder
    try:
        folder.rmdir()
    except OSError:
        pass

def main():
    dst_dir = Path.cwd()

    if len(sys.argv) < 2:
        print("Usage: python script.py folder1 folder2 ...")
        sys.exit(1)

    for arg in sys.argv[1:]:
        move_folder_contents(Path(arg), dst_dir)

if __name__ == "__main__":
    main()
```

## Scrape trial.spmpaper.me

This is highly specific, and likely won't be useful to anybody except for a few people. This Python script scrapes the trial.spmpaper.me site, which contains past-year questions for SPM (Sijil Pelajaran Malaysia, or in English it's Certificate of Education of Malaysia). Here is the script:

```python
import os
import time
import requests
from bs4 import BeautifulSoup
from urllib.parse import urljoin, urlparse, parse_qs, unquote

# --- CONFIGURATION ---
# Base URL and the starting point (Root directory)
BASE_URL = "https://trial.spmpaper.me/"
START_URL = "https://trial.spmpaper.me/"
# Main directory to save the massive archive
DOWNLOAD_FOLDER = "SPM_Complete_Archive"

# Track visited directories to prevent infinite loops
visited_dirs = set()

def sanitize_folder_name(path_param):
    """Converts a URL dir parameter into a safe local folder structure."""
    if not path_param:
        return ""
    decoded_path = unquote(path_param)
    # Remove characters that are illegal in Windows/Linux file systems
    for char in ['<', '>', ':', '"', '/', '\\', '|', '?', '*']:
        decoded_path = decoded_path.replace(char, '_')
    return decoded_path

def scrape_directory(url):
    """Recursively crawls all directories on the site and downloads all PDFs."""
    parsed_url = urlparse(url)
    query_params = parse_qs(parsed_url.query)
    current_dir_param = query_params.get('dir', [''])[0]
    
    # Prevent scanning the same folder twice
    if current_dir_param in visited_dirs:
        return
    
    print(f"\n[Scanning Directory]: {unquote(current_dir_param) if current_dir_param else 'Root'}")
    visited_dirs.add(current_dir_param)
    
    try:
        response = requests.get(url, timeout=15)
        response.raise_for_status()
    except Exception as e:
        print(f"Failed to fetch {url}: {e}")
        return

    soup = BeautifulSoup(response.text, 'html.parser')
    
    # Set local folder destination
    local_subfolder = sanitize_folder_name(current_dir_param)
    target_dir = os.path.join(DOWNLOAD_FOLDER, local_subfolder)
    os.makedirs(target_dir, exist_ok=True)
    
    subfolders_to_visit = []
    
    # Find all links on the current page
    for anchor in soup.find_all('a', href=True):
        href = anchor['href']
        full_url = urljoin(BASE_URL, href)
        
        parsed_href = urlparse(full_url)
        href_queries = parse_qs(parsed_href.query)
        dir_param = href_queries.get('dir', [''])[0]
        
        # Case 1: It's a directory link
        if '?dir=' in href:
            # Avoid going backwards to a parent directory (ensure we are moving forward)
            # Root directory is empty string '', so everything moving forward is longer than it
            if len(dir_param) > len(current_dir_param) and dir_param.startswith(current_dir_param):
                if full_url not in visited_dirs:
                    subfolders_to_visit.append(full_url)
                
        # Case 2: It's a PDF link
        elif href.lower().endswith('.pdf'):
            filename = os.path.basename(parsed_href.path)
            if not filename:
                filename = href.split('/')[-1]
                
            file_save_path = os.path.join(target_dir, unquote(filename))
            
            # Download file if it does not exist locally
            if not os.path.exists(file_save_path):
                print(f" -> Downloading: {unquote(filename)}")
                try:
                    file_res = requests.get(full_url, timeout=30)
                    with open(file_save_path, 'wb') as f:
                        f.write(file_res.content)
                    time.sleep(0.4)  # Politeness delay to avoid hitting rate-limits
                except Exception as e:
                    print(f"    [ERROR] Could not download {filename}: {e}")
            else:
                print(f" -> Skipping (Already Exists): {unquote(filename)}")

    # Recursively visit every subdirectory discovered
    for subfolder in subfolders_to_visit:
        scrape_directory(subfolder)

if __name__ == "__main__":
    print("WARNING: Starting Global Site Scraper. This will download ALL available subjects.")
    start_time = time.time()
    
    scrape_directory(START_URL)
    
    print("\n==========================================")
    print(f"Finished! Entire site downloaded to: '{DOWNLOAD_FOLDER}'")
    print(f"Total time taken: {round(time.time() - start_time, 2)} seconds.")
    print("==========================================")
```

This script is generated by Gemini Flash 3.5.
